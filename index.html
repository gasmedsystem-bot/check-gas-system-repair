⚡ SQL Index — เพิ่มความเร็ว

คัดลอก SQL ไปรันใน Supabase SQL Editor

CREATE INDEX IF NOT EXISTS idx_ir_sys_date
  ON inspection_records(system_type, recorded_at DESC);

CREATE INDEX IF NOT EXISTS idx_ir_status
  ON inspection_records(status);

CREATE INDEX IF NOT EXISTS idx_ir_building
  ON inspection_records(system_type, building);

CREATE INDEX IF NOT EXISTS idx_ir_sys_status
  ON inspection_records(system_type, status)
  WHERE status = 'alert';
